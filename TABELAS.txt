CREATE TABLE public.usuario (
	id_usuario int4 NOT NULL,
	nome varchar(50) NOT NULL,
	email varchar(50) NOT NULL,
	ativo bool NOT NULL DEFAULT true,
	CONSTRAINT id_usuario PRIMARY KEY (id_usuario)
);
